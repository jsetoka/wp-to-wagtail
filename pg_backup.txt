#!/usr/bin/env bash
set -euo pipefail

DB_HOST="localhost"
DB_PORT="5432"
DB_NAME="oic_db"
DB_USER="oic_user"
BACKUP_DIR="/var/backups/postgres"
TIMESTAMP=$(date +%F_%H-%M-%S)
BACKUP_FILE="${BACKUP_DIR}/${DB_NAME}_${TIMESTAMP}.dump"

mkdir -p "$BACKUP_DIR"

export PGPASSWORD="mot_de_passe"

pg_dump \
  -h "$DB_HOST" \
  -p "$DB_PORT" \
  -U "$DB_USER" \
  -d "$DB_NAME" \
  -Fc \
  -f "$BACKUP_FILE"

find "$BACKUP_DIR" -type f -name "${DB_NAME}_*.dump" -mtime +7 -delete

echo "Backup créé : $BACKUP_FILE"
